async function deleteWithLog(tableName, deleteFunction) {
    try {
        const result = await deleteFunction();
        const count = result.count || 0;
        console.log(`   ‚úì ${tableName}: ${count} registro(s) removido(s)`);
        return count;
    } catch (error) {
        console.error(`   ‚ùå Erro ao deletar ${tableName}:`, error.message);
        throw error;
    }
}
